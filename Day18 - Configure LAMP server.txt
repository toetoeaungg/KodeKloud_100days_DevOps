## Install httpd and change port 6300
sudo yum install -y httpd php php-mysqlnd php-cli php-gd php-mbstring php-xml php-pear
sudo systemctl enable httpd
sudo systemctl start httpd

sudo vi /etc/httpd/conf/httpd.conf

## change listen port 80 to ->> port 6300

## verify running service in which port
sudo netstat -tulpn | grep 6300

## Install MariaDB in DB server

sudo yum update
sudo yum install -y mariadb-server mariadb

sudo systemctl enable mariadb
sudo systemctl start mariadb
sudo systemctl status mariadb   # Optional, check service status

## Secure MariaDB:
sudo mysql_secure_installation
